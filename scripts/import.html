API import geocode

I made an API request to openstreetmap to geocode NBA city coordinates

# API request
geo_req <- request("https://nominatim.openstreetmap.org/search") |>
  req_url_query(
    q = "Boston, MA",
    format = "json",
    limit = 1
  ) |>
  req_user_agent("DSC210-NBA-Project")

# request
geo_resp <- geo_req |> req_perform()

# parse 
geo_data <- geo_resp |>
  resp_body_string() |>
  fromJSON()

wbc_url <- "https://www.worldsbestcities.com/rankings/americas-best-cities/"
wbc_page <- read_html(wbc_url)

# pull text
ranking_text <- wbc_page |>
  html_elements("#rankings") |>
  html_text2()

# split
city_blocks <- ranking_text |>
  str_split("\n\\d+\\. ")

# make tibble
cities_tbl <- tibble(
  raw_text = city_blocks[[1]][-1])

# extract state data
cities_tbl <- cities_tbl |>
  mutate(
    # city + state
    city_state = str_extract(raw_text, "^[A-Za-z .'-]+, [A-Z]{2}"),
    city = str_remove(city_state, ", [A-Z]{2}$"),
    state = str_extract(city_state, "[A-Z]{2}$"),
    
    # ranking
    ranking = row_number(),
    # population
    population = str_extract(raw_text, "Metro: [0-9,]+") |>
      str_remove("Metro: ") |>
      str_remove_all(",") |>
      as.numeric(),

    # ranking number
    highlighted_rank_1 = str_extract(
      raw_text,
      "Highlighted Rankings\\s*\\n\\s*#\\d+"
    ) |>
      str_extract("#\\d+") |>
      str_remove("#") |>
      as.numeric(),

    # top two categories
    highlighted_category_1 = str_extract(
      raw_text,
      "Highlighted Rankings[\\s\\S]*?#\\d+\\n([^\\n]+)"
    ) |>
      str_remove("Highlighted Rankings[\\s\\S]*?#\\d+\\n") |>
      str_trim(),

    highlighted_category_2 = str_extract(
      raw_text,
      "Highlighted Rankings[\\s\\S]*?#\\d+\\n[^\\n]+\\n#\\d+\\n([^\\n]+)"
    ) |>
      str_remove("Highlighted Rankings[\\s\\S]*?#\\d+\\n[^\\n]+\\n#\\d+\\n") |>
      str_trim()
  ) |>
  select(
    ranking,
    city,
    state,
    population,
    highlighted_rank_1,
    highlighted_category_1,
    highlighted_category_2,
    raw_text
  )
